/*商品表*/
create table productTable									
(									
	pro_id number(6) primary key								
	,pro_name  VARCHAR2(100) not null							
	,pro_money number(7) not null								
	,pro_stock  number(6) default 0 not null						
	,pro_calorie number(6) CONSTRAINTS ck_pro_calorie CHECK (pro_calorie >= 0)			
	,pro_release_date DATE default sysdate	
	,pro_type VARCHAR2(8) not null								
	,pro_image VARCHAR2(100)								
	,pro_protein number(4) default 0
	,pro_carbohydrate number(4) default 0
	,pro_lipid number(4) default 0								
	,pro_vitamin number(4) default 0
	,pro_inorganic number(4) default 0
	,pro_wheat number(1) default 0 not null CONSTRAINTS ck_pro_wheat CHECK (pro_wheat IN (0,1))								
	,pro_egg number(1) default 0 not null CONSTRAINTS ck_pro_egg CHECK (pro_egg IN (0,1))								
	,pro_milk number(1) default 0 not null CONSTRAINTS ck_pro_milk CHECK (pro_milk IN (0,1))								
	,pro_peanuts number(1) default 0 not null CONSTRAINTS ck_pro_peanuts CHECK (pro_peanuts IN (0,1))								,pro_buckwheat number(1) default 0 not null CONSTRAINTS ck_pro_buckwheat CHECK (pro_buckwheat IN (0,1))								,pro_shrimp number(1) default 0 not null CONSTRAINTS ck_pro_shrimp CHECK (pro_shrimp IN (0,1))								
	,pro_club number(1) default 0 not null CONSTRAINTS ck_pro_club CHECK (pro_club IN (0,1))								
);									
										
CREATE SEQUENCE sq_pro_id;

/*ユーザー表*/										
create table userTable										
(										
	user_id number primary key									
	,user_lastname VARCHAR2(50) not null									
	,user_firstname VARCHAR2(50) not null									
	,user_lastname_hira VARCHAR2(80) not null									
	,user_firstname_hira VARCHAR2(80) not null									
	,user_sex VARCHAR2(3) CONSTRAINTS ck_user_sex CHECK (user_sex IN ('男', '女'))										
	,user_mail VARCHAR2(256)										
	,user_phone VARCHAR2(11)										
	,user_address_id number(8)										
	,user_pass VARCHAR2(16)										
	,user_birthday VARCHAR2(4)										
	,user_admin number(1) default 0 not null									
	,CONSTRAINT fk_user_address_id foreign key(user_address_id) references  addressTable(address_id)									
);										
										
CREATE SEQUENCE sq_user_id;

/*注文表*/										
create table orderTable										
(										
	or_id number primary key									
	,or_pro_id number(6)									
	,or_user_id number(6) not null									
	,or_count number(2) CONSTRAINTS ck_or_count CHECK ( or_count BETWEEN 0 AND 99)									
	,or_type number(1) not null CONSTRAINTS ck_or_type CHECK (or_type IN ('0', '1'))									
	,or_address_id number(8) not null										
	,or_date DATE default sysdate										
	,or_store number(1) default 0 not null										
	,CONSTRAINT fk_or_pro_id foreign key(or_pro_id) references  productTable(pro_id)										,CONSTRAINT fk_or_user_id foreign key(or_user_id) references  userTable(user_id)										,CONSTRAINT fk_or_address_id foreign key(or_address_id) references  addressTable(address_id)									
);										
										
CREATE SEQUENCE sq_or_id;
